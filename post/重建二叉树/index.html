剑指 Offer 07. 重建二叉树

难度中等181

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
      if (preorder.empty() || inorder.empty()) {
        return nullptr;
      }
      if(preorder.size() != inorder.size()) {
        return nullptr;
      }
      unordered_map<int, int> mp;
      for (int i = 0; i < inorder.size(); ++i) {
        mp[inorder[i]] = i;
      }
      return buildTree(preorder, mp, 0, 0, inorder.size() - 1);
    }
    TreeNode* buildTree(vector<int>& preorder, unordered_map<int, int>& mp, int pre_root, int in_start, int in_end) {
      if (in_start > in_end) {
        return nullptr;
      }
      TreeNode* root = new TreeNode(preorder[pre_root]);
      int idx = mp[preorder[pre_root]];
      root->left = buildTree(preorder, mp, pre_root + 1, in_start, idx - 1);
      root->right = buildTree(preorder, mp, pre_root + 1 + idx - in_start, idx + 1, in_end);
      return root;
    }
};
