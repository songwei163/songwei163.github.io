一只青蛙一次可以跳上1级台阶，也可以跳上2级。求该青蛙跳上一个n级的台阶总共有多少种跳法（先后次序不同算不同的结果）

class Solution {
public:
    int jumpFloor(int n) {
      if (n <= 1) {
        return 1;
      }
      int first = 1;
      int second = 1;
      int res = 0;
      for (int i = 2; i < n + 1; ++i) {
        res = first + second;
        first = second;
        second = res;
      }
      return res;
    }
};
